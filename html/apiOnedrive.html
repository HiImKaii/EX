// Cấu hình Microsoft Authentication const msalConfig = { auth: { clientId: 'a8bd8006-9ec3-4d14-a4a1-86ef79a8f226', // Thay bằng Application ID từ Azure Portal authority: 'https://login.microsoftonline.com/common', redirectUri: window.location.origin, // URL hiện tại của ứng dụng }, cache: { cacheLocation: 'localStorage', storeAuthStateInCookie: false } }; // Phạm vi quyền truy cập OneDrive const scopes = [ 'files.readwrite', 'user.read' ]; let msalInstance = null; let account = null; // Khởi tạo MSAL function initMSAL() { try { // Import thư viện MSAL const script = document.createElement('script'); script.src = 'https://alcdn.msauth.net/browser/2.30.0/js/msal-browser.min.js'; script.onload = () => { msalInstance = new msal.PublicClientApplication(msalConfig); // Kiểm tra nếu người dùng đã đăng nhập const currentAccounts = msalInstance.getAllAccounts(); if (currentAccounts.length > 0) { account = currentAccounts[0]; updateOneDriveButtons(true); } else { updateOneDriveButtons(false); } }; document.head.appendChild(script); } catch (error) { console.error('Lỗi khởi tạo MSAL:', error); } } // Đăng nhập vào Microsoft async function signInToMicrosoft() { if (!msalInstance) { alert('Đang tải thư viện Microsoft Authentication, vui lòng thử lại sau vài giây'); initMSAL(); return; } try { const loginResponse = await msalInstance.loginPopup({ scopes: scopes, prompt: 'select_account' }); account = loginResponse.account; updateOneDriveButtons(true); } catch (error) { console.error('Lỗi đăng nhập:', error); alert('Không thể đăng nhập: ' + error); } } // Đăng xuất async function signOutOfMicrosoft() { if (!msalInstance) return; try { await msalInstance.logoutPopup(); account = null; updateOneDriveButtons(false); } catch (error) { console.error('Lỗi đăng xuất:', error); } } // Lấy token truy cập async function getAccessToken() { if (!msalInstance || !account) { throw new Error('Chưa đăng nhập'); } try { const silentRequest = { scopes: scopes, account: account, forceRefresh: false }; const silentResult = await msalInstance.acquireTokenSilent(silentRequest); return silentResult.accessToken; } catch (error) { console.error('Lỗi lấy token:', error); // Nếu token hết hạn, thử lấy lại if (error instanceof msal.InteractionRequiredAuthError) { const interactionResult = await msalInstance.acquireTokenPopup(silentRequest); return interactionResult.accessToken; } throw error; } } // Lưu dữ liệu lên OneDrive async function saveToOneDrive() { try { if (!account) { await signInToMicrosoft(); if (!account) return; // Người dùng hủy đăng nhập } const accessToken = await getAccessToken(); const dataStr = JSON.stringify(data); const fileName = 'cay-kien-thuc-' + new Date().toISOString().slice(0, 10) + '.json'; // Hiển thị trạng thái showStatus('Đang lưu lên OneDrive...'); // Tạo upload session cho file lớn const createSessionResponse = await fetch( 'https://graph.microsoft.com/v1.0/me/drive/root:/' + fileName + ':/createUploadSession', { method: 'POST', headers: { 'Authorization': 'Bearer ' + accessToken, 'Content-Type': 'application/json' }, body: JSON.stringify({ "@microsoft.graph.conflictBehavior": "replace" }) } ); if (!createSessionResponse.ok) { throw new Error('Không thể tạo phiên tải lên: ' + await createSessionResponse.text()); } const sessionData = await createSessionResponse.json(); const uploadUrl = sessionData.uploadUrl; // Upload file const uploadResponse = await fetch(uploadUrl, { method: 'PUT', headers: { 'Content-Type': 'application/json', }, body: dataStr }); if (!uploadResponse.ok) { throw new Error('Lỗi khi tải lên: ' + await uploadResponse.text()); } hideStatus(); alert('Đã lưu dữ liệu lên OneDrive thành công!'); } catch (error) { console.error('Lỗi lưu OneDrive:', error); hideStatus(); alert('Không thể lưu lên OneDrive: ' + error.message); } } // Tải dữ liệu từ OneDrive async function loadFromOneDrive() { try { if (!account) { await signInToMicrosoft(); if (!account) return; // Người dùng hủy đăng nhập } const accessToken = await getAccessToken(); // Hiển thị trạng thái showStatus('Đang tải danh sách tệp từ OneDrive...'); // Lấy danh sách file JSON const filesResponse = await fetch( 'https://graph.microsoft.com/v1.0/me/drive/root/search(q=\'.json\')?select=name,id,lastModifiedDateTime', { headers: { 'Authorization': 'Bearer ' + accessToken } } ); if (!filesResponse.ok) { throw new Error('Không thể lấy danh sách tệp: ' + await filesResponse.text()); } const filesData = await filesResponse.json(); hideStatus(); // Lọc chỉ các tệp cây kiến thức const knowledgeFiles = filesData.value.filter(file => file.name.toLowerCase().includes('cay-kien-thuc') || file.name.toLowerCase().includes('knowledge') ); if (knowledgeFiles.length === 0) { alert('Không tìm thấy tệp dữ liệu cây kiến thức nào trên OneDrive của bạn'); return; } // Sắp xếp theo thời gian sửa đổi mới nhất knowledgeFiles.sort((a, b) => new Date(b.lastModifiedDateTime) - new Date(a.lastModifiedDateTime) ); // Hiển thị dialog chọn file showFileSelectionDialog(knowledgeFiles, async (fileId) => { try { showStatus('Đang tải dữ liệu...'); // Tải nội dung file const fileResponse = await fetch( `https://graph.microsoft.com/v1.0/me/drive/items/${fileId}/content`, { headers: { 'Authorization': 'Bearer ' + accessToken } } ); if (!fileResponse.ok) { throw new Error('Không thể tải tệp: ' + await fileResponse.text()); } const fileContent = await fileResponse.text(); const importedData = JSON.parse(fileContent); // Kiểm tra tính hợp lệ của dữ liệu if (typeof importedData !== 'object') { throw new Error('Dữ liệu không hợp lệ'); } // Xác nhận ghi đè if (confirm('Dữ liệu hiện tại sẽ bị ghi đè. Bạn có chắc chắn muốn tiếp tục?')) { data = importedData; saveData(); goHome(); alert('Tải dữ liệu từ OneDrive thành công!'); } hideStatus(); } catch (error) { console.error('Lỗi tải tệp:', error); hideStatus(); alert('Không thể tải tệp: ' + error.message); } }); } catch (error) { console.error('Lỗi tải từ OneDrive:', error); hideStatus(); alert('Không thể tải từ OneDrive: ' + error.message); } } // Hiển thị dialog chọn file function showFileSelectionDialog(files, onSelect) { // Tạo dialog const dialog = document.createElement('div'); dialog.className = 'dialog'; dialog.style.width = '500px'; // Tiêu đề const title = document.createElement('h3'); title.textContent = 'Chọn tệp từ OneDrive'; dialog.appendChild(title); // Danh sách file const fileList = document.createElement('div'); fileList.style.maxHeight = '300px'; fileList.style.overflowY = 'auto'; fileList.style.margin = '15px 0'; fileList.style.border = '1px solid var(--border-color)'; fileList.style.borderRadius = '5px'; fileList.style.padding = '5px'; files.forEach(file => { const fileItem = document.createElement('div'); fileItem.style.padding = '10px'; fileItem.style.margin = '5px 0'; fileItem.style.borderRadius = '5px'; fileItem.style.backgroundColor = 'var(--card-bg)'; fileItem.style.cursor = 'pointer'; // Format ngày const date = new Date(file.lastModifiedDateTime); const formattedDate = `${date.getDate()}/${date.getMonth()+1}/${date.getFullYear()} ${date.getHours()}:${date.getMinutes().toString().padStart(2, '0')}`; fileItem.innerHTML = `
${file.name}
Cập nhật: ${formattedDate}
`; fileItem.addEventListener('click', () => { document.body.removeChild(dialog); onSelect(file.id); }); fileList.appendChild(fileItem); }); dialog.appendChild(fileList); // Nút hủy const actions = document.createElement('div'); actions.className = 'dialog-actions'; const cancelBtn = document.createElement('button'); cancelBtn.textContent = 'Hủy'; cancelBtn.onclick = () => document.body.removeChild(dialog); actions.appendChild(cancelBtn); dialog.appendChild(actions); // Hiển thị dialog document.body.appendChild(dialog); } // Cập nhật nút OneDrive function updateOneDriveButtons(isLoggedIn) { // Thêm hoặc cập nhật nút OneDrive const mainControls = document.querySelector('#mainView .controls'); const subControls = document.querySelector('#subView .controls'); // Tạo các nút const saveBtn = document.createElement('button'); saveBtn.innerHTML = '☁️ Lưu lên OneDrive'; saveBtn.onclick = saveToOneDrive; const loadBtn = document.createElement('button'); loadBtn.innerHTML = '☁️ Tải từ OneDrive'; loadBtn.onclick = loadFromOneDrive; const loginBtn = document.createElement('button'); loginBtn.innerHTML = isLoggedIn ? '🔓 Đăng xuất OneDrive' : '🔑 Đăng nhập OneDrive'; loginBtn.onclick = isLoggedIn ? signOutOfMicrosoft : signInToMicrosoft; // Thêm vào trang chính if (mainControls) { // Xóa nút cũ nếu có const oldSaveBtn = mainControls.querySelector('button[data-onedrive="save"]'); const oldLoadBtn = mainControls.querySelector('button[data-onedrive="load"]'); const oldLoginBtn = mainControls.querySelector('button[data-onedrive="login"]'); if (oldSaveBtn) mainControls.removeChild(oldSaveBtn); if (oldLoadBtn) mainControls.removeChild(oldLoadBtn); if (oldLoginBtn) mainControls.removeChild(oldLoginBtn); // Thêm nút mới if (isLoggedIn) { saveBtn.setAttribute('data-onedrive', 'save'); loadBtn.setAttribute('data-onedrive', 'load'); mainControls.appendChild(saveBtn); mainControls.appendChild(loadBtn); } loginBtn.setAttribute('data-onedrive', 'login'); mainControls.appendChild(loginBtn); } // Thêm vào trang con if (subControls) { // Xóa nút cũ nếu có const oldSaveBtn = subControls.querySelector('button[data-onedrive="save"]'); const oldLoadBtn = subControls.querySelector('button[data-onedrive="load"]'); const oldLoginBtn = subControls.querySelector('button[data-onedrive="login"]'); if (oldSaveBtn) subControls.removeChild(oldSaveBtn); if (oldLoadBtn) subControls.removeChild(oldLoadBtn); if (oldLoginBtn) subControls.removeChild(oldLoginBtn); // Thêm nút mới if (isLoggedIn) { const saveBtnClone = saveBtn.cloneNode(true); saveBtnClone.onclick = saveToOneDrive; saveBtnClone.setAttribute('data-onedrive', 'save'); const loadBtnClone = loadBtn.cloneNode(true); loadBtnClone.onclick = loadFromOneDrive; loadBtnClone.setAttribute('data-onedrive', 'load'); subControls.appendChild(saveBtnClone); subControls.appendChild(loadBtnClone); } const loginBtnClone = loginBtn.cloneNode(true); loginBtnClone.onclick = isLoggedIn ? signOutOfMicrosoft : signInToMicrosoft; loginBtnClone.setAttribute('data-onedrive', 'login'); subControls.appendChild(loginBtnClone); } } // Hiển thị trạng thái function showStatus(message) { // Kiểm tra nếu đã có status box let statusBox = document.getElementById('status-box'); if (!statusBox) { statusBox = document.createElement('div'); statusBox.id = 'status-box'; statusBox.style.position = 'fixed'; statusBox.style.bottom = '20px'; statusBox.style.right = '20px'; statusBox.style.padding = '10px 20px'; statusBox.style.backgroundColor = 'var(--button-bg)'; statusBox.style.color = 'white'; statusBox.style.borderRadius = '5px'; statusBox.style.zIndex = '9999'; statusBox.style.boxShadow = '0 0 10px rgba(0,0,0,0.2)'; document.body.appendChild(statusBox); } statusBox.textContent = message; statusBox.style.display = 'block'; } // Ẩn trạng thái function hideStatus() { const statusBox = document.getElementById('status-box'); if (statusBox) { statusBox.style.display = 'none'; } } // Thêm CSS cho các thành phần mới function addOneDriveStyles() { const style = document.createElement('style'); style.textContent = ` .file-item { padding: 10px; margin: 5px 0; background-color: var(--card-bg); border-radius: 5px; cursor: pointer; } .file-item:hover { background-color: rgba(104, 213, 155, 0.1); } #status-box { transition: all 0.3s ease; } `; document.head.appendChild(style); } // Khởi tạo OneDrive integration function initOneDriveIntegration() { addOneDriveStyles(); // Thêm nút vào controls const controls = document.createElement('div'); controls.className = 'controls'; if (!document.querySelector('#mainView .controls')) { document.getElementById('mainView').insertBefore(controls, document.getElementById('mainView').firstChild); } // Khởi tạo MSAL sau khi DOM đã tải setTimeout(initMSAL, 500); } // Gọi hàm này trong initApp() để khởi tạo OneDrive function initApp() { try { // Tải chủ đề const savedTheme = localStorage.getItem('knowledgeTreeTheme'); if (savedTheme === 'dark') { document.body.classList.add('dark-theme'); currentTheme = 'dark'; } // Tải dữ liệu loadData(); renderMain(); // Khởi tạo OneDrive initOneDriveIntegration(); } catch (error) { console.error('Lỗi khởi tạo ứng dụng:', error); alert('Đã xảy ra lỗi khi khởi tạo ứng dụng. Dữ liệu sẽ được thiết lập lại.'); resetToDefaultData(); renderMain(); initOneDriveIntegration(); // Vẫn khởi tạo OneDrive } }
