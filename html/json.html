// Thêm các nút xuất/nhập vào phần controls function updateControls() { const mainControls = document.createElement('div'); mainControls.className = 'controls'; mainControls.innerHTML = ` `; // Thêm vào trang chính const mainView = document.getElementById('mainView'); const existingControls = mainView.querySelector('.controls'); if (existingControls) { existingControls.innerHTML = mainControls.innerHTML; } else { mainView.insertBefore(mainControls, mainView.firstChild); } // Thêm nút xuất vào các trang con const subControls = document.querySelector('#subView .controls'); if (subControls) { const exportBtn = document.createElement('button'); exportBtn.innerHTML = '💾 Xuất dữ liệu'; exportBtn.onclick = exportData; subControls.appendChild(exportBtn); const importBtn = document.createElement('button'); importBtn.innerHTML = '📂 Nhập dữ liệu'; importBtn.onclick = importData; subControls.appendChild(importBtn); } } // Hàm xuất dữ liệu function exportData() { try { const dataStr = JSON.stringify(data, null, 2); const blob = new Blob([dataStr], { type: 'application/json' }); const url = URL.createObjectURL(blob); const a = document.createElement('a'); a.href = url; a.download = 'cay-kien-thuc-' + new Date().toISOString().slice(0, 10) + '.json'; document.body.appendChild(a); a.click(); setTimeout(() => { document.body.removeChild(a); URL.revokeObjectURL(url); }, 100); } catch (error) { console.error('Lỗi khi xuất dữ liệu:', error); alert('Không thể xuất dữ liệu: ' + error.message); } } // Hàm nhập dữ liệu function importData() { const input = document.createElement('input'); input.type = 'file'; input.accept = 'application/json'; input.onchange = function(e) { const file = e.target.files[0]; if (!file) return; const reader = new FileReader(); reader.onload = function(e) { try { const importedData = JSON.parse(e.target.result); // Kiểm tra tính hợp lệ của dữ liệu if (typeof importedData !== 'object') { throw new Error('Dữ liệu không hợp lệ'); } // Xác nhận ghi đè if (confirm('Dữ liệu hiện tại sẽ bị ghi đè. Bạn có chắc chắn muốn tiếp tục?')) { data = importedData; saveData(); goHome(); alert('Nhập dữ liệu thành công!'); } } catch (error) { console.error('Lỗi khi đọc tệp:', error); alert('Tệp không hợp lệ hoặc bị hỏng: ' + error.message); } }; reader.readAsText(file); }; input.click(); } // Thêm hàm gọi này vào cuối hàm initApp() function initApp() { try { // Tải chủ đề const savedTheme = localStorage.getItem('knowledgeTreeTheme'); if (savedTheme === 'dark') { document.body.classList.add('dark-theme'); currentTheme = 'dark'; } // Tải dữ liệu loadData(); renderMain(); // Thêm dòng này updateControls(); } catch (error) { console.error('Lỗi khởi tạo ứng dụng:', error); alert('Đã xảy ra lỗi khi khởi tạo ứng dụng. Dữ liệu sẽ được thiết lập lại.'); resetToDefaultData(); renderMain(); updateControls(); // Thêm dòng này } }
